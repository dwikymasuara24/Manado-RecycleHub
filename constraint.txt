CREATE TABLE `survey_responses` (
  `id` bigint unsigned NOT NULL AUTO_INCREMENT,
  `response_code` varchar(20) COLLATE utf8mb4_general_ci DEFAULT NULL,
  `nama` varchar(150) COLLATE utf8mb4_general_ci DEFAULT NULL,
  `email` varchar(191) COLLATE utf8mb4_general_ci DEFAULT NULL,
  `nomor_wa` varchar(30) COLLATE utf8mb4_general_ci DEFAULT NULL,
  `alamat` text COLLATE utf8mb4_general_ci,
  `user_id` bigint unsigned DEFAULT NULL,
  `q1_sampah_mendesak` enum('Ya','Tidak') COLLATE utf8mb4_general_ci DEFAULT NULL,
  `q2_paham_3r` enum('Ya','Tidak') COLLATE utf8mb4_general_ci DEFAULT NULL,
  `q3_daur_ulang_rumah` enum('Ya','Tidak') COLLATE utf8mb4_general_ci DEFAULT NULL,
  `q4_pilah_organik_anorganik` enum('Ya','Tidak') COLLATE utf8mb4_general_ci DEFAULT NULL,
  `q5_jenis_sampah_didaur_ulang` longtext CHARACTER SET utf8mb4 COLLATE utf8mb4_bin,
  `q6_kesulitan_daur_ulang` text COLLATE utf8mb4_general_ci,
  `q7_bersedia_pilah` enum('Ya','Tidak') COLLATE utf8mb4_general_ci DEFAULT NULL,
  `ip_address` varchar(45) COLLATE utf8mb4_general_ci DEFAULT NULL,
  `user_agent` varchar(500) COLLATE utf8mb4_general_ci DEFAULT NULL,
  `session_id` varchar(191) COLLATE utf8mb4_general_ci DEFAULT NULL,
  `status` enum('selesai','tidak_selesai') COLLATE utf8mb4_general_ci NOT NULL DEFAULT 'selesai',
  `created_at` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `updated_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `q6_kesulitan` text COLLATE utf8mb4_general_ci,
  PRIMARY KEY (`id`),
  KEY `idx_surv_user` (`user_id`),
  KEY `idx_surv_email` (`email`),
  CONSTRAINT `fk_surv_user` FOREIGN KEY (`user_id`) REFERENCES `users` (`id`) ON DELETE SET NULL,
  CONSTRAINT `survey_responses_chk_1` CHECK (json_valid(`q5_jenis_sampah_didaur_ulang`))
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci COMMENT='Jawaban kuesioner warga'
